vector<int> solution(vector<int> numbers) {
    vector<int> answer(numbers.size(), -1);
    stack<int> s;
    s.push(0);
    for (int i = 1; i < numbers.size(); ++i)
    {
        while (!s.empty() && numbers[s.top()] < numbers[i])
        {
            answer[s.top()] = numbers[i];
            s.pop();
        }
        s.push(i);
    }
        
    return answer;
}
